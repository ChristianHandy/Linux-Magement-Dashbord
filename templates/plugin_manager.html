Write Your Own Plugin

Create a .py file with an addon_meta dict and an optional register(app, core) function. Upload it here to activate it instantly. 

addon_meta = {
    "name":        "My Plugin",
    "version":     "1.0.0",
    "author":      "Your Name",
    "description": "What this plugin does",
    "scope":       "global",   # global | disks | hosts | dashboard | scanner

    # ── Hook registrations (pick any) ───────────────────────────────────────
    # Render HTML on the Home dashboard
    "dashboard_widget": lambda hosts, stats: "<div class='card'>Hello!</div>",

    # Add action buttons to each host card
    "host_card_action": lambda host: f"<a href='/ping/{host['name']}'>Ping</a>",

    # Add a tab to the host detail view
    "host_detail_tab": lambda host: {"tab_label": "SSH Keys", "tab_content": "<p>...</p>"},

    # Add a link to the sidebar
    "sidebar_link": lambda: {"label": "My Tool", "url": "/my_tool", "icon": "🔧"},

    # Called after every update run
    "update_post": lambda host, result: notify_slack(host, result),

    # Add buttons to scanner results
    "scanner_action": lambda ip, info: f"<a href='/ssh/{ip}'>Connect</a>",
}

def register(app, core):
    from flask import Blueprint, render_template
    bp = Blueprint("my_plugin", __name__)

    @bp.route("/my_tool")
    def my_tool():
        return "<h1>My Plugin Page</h1>"

    app.register_blueprint(bp)
